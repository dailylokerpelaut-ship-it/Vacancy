async function loadJobs(){ try{ const url=`https://docs.google.com/spreadsheets/d/${SHEET_ID}/gviz/tq?tqx=out:json&sheet=${encodeURIComponent(SHEET_TAB)}`; const res=await fetch(url); const txt=await res.text(); const json=JSON.parse(txt.substring(47).slice(0,-2)); const rows=json.table.rows; const parsed=rows .filter(r=>r.c[1]?.v?.toString().toUpperCase()==='APPROVED') .map((r,i)=>{ const ts=parseTimestamp(r.c[0]?.v); const posRaw=r.c[6]?.v?.toString()||''; const reqRaw=r.c[5]?.v?.toString()||''; const posList=posRaw.split(',').map(p=>p.trim()).filter(Boolean); const reqList=reqRaw .split(/\n|\||;/) .map(s=>s.replace(/^[•\-–]\s*/,'').trim()) .filter(Boolean); const positions=posList.length ?posList.map(p=>({title:p,reqs:reqList})) :[{title:'Lihat deskripsi',reqs:reqList}]; return{ id:i, urgent:r.c[2]?.v?.toString().toLowerCase()==='ya', featured:r.c[10]?.v?.toString().toLowerCase()==='ya', company:r.c[9]?.v?.toString()||'', vessel:r.c[8]?.v?.toString()||'', trade:r.c[7]?.v?.toString()||'International', positions, contact:r.c[4]?.v?.toString()||'', free:r.c[3]?.v?.toString().toLowerCase()!=='tidak', desc:r.c[11]?.v?.toString()||'', timestamp:ts }; }) .filter(j=>j.company && !isExpired(j.timestamp)); allJobs=parsed.length>0?parsed:DEMO; }catch(e){ console.warn('Sheet load failed:',e); allJobs=DEMO; } render(); }
